The figure below shows a rigid spool \basis{B} in contact with the ground
\basis{N}. Denote the spool's inner radius as $r$ and outer radius as $R$.
The point of the spool in contact with the ground is $B_N$.
In completing this problem, please introduce appropriate unit vectors,
identifiers (labels), and anything else you need to help you.
\begin{center}
  \begin{minipage}[t]{0.44\linewidth}
    \minipageTopAnchor
    \flushleft
    %\vspace*{-3cm}
    \includegraphicsAB[width=1\linewidth]
      {spool-solution.png}
      {spool.png}
  \end{minipage}
\end{center}
\begin{enumerate}
  \item Calculate \moment{\bvec{F}_T/B_N}, the moment due to force \force{T}
    about $B_N$, in terms of $r$, $R$, $\theta$, and any other quantities necessary.
    \\[0.45pc]
    \Solution{}{0.53\linewidth}{
      \begin{minipage}[t]{0.99\linewidth}
        \begin{tabular}{lcl}
          \moment{\bvec{F}_T/B_N} & $\deff$ & $\posvec{B_N}{B_I}~ \CrossProduct ~\bvec{F}_T$
          \\[1.0pc]
          &=&$(R~\uvecy{n}-r~\uvecy{b})~ \CrossProduct ~(F_T~\uvecx{b})$
          \\[1.0pc]
          &=&$F_T R~\uvecy{n}~ \CrossProduct~\uvecx{b}\minus F_T r~\uvecy{b}~ \CrossProduct~\uvecx{b}$
          \\[1.0pc]
          &=&$F_T R(\minus~\uvecz{n})~\sin(90\minus \theta) \minus F_T r(\minus~\uvecz{n})$
          \\[1.0pc]
          &=&$F_T(r\minus R \cos\theta)~\uvecz{n}$
          \\[1.0pc]
          &=&$10N \left(\SI{1}{m}\minus\SI{2}{m}\left(\fracDisplaystyle{1}{2}\right)\right)~\uvecz{n}$
          \\[1.0pc]
          &=&$0~\uvecz{n}$
          \\[1.0pc]
        \end{tabular}
      \end{minipage}
    }
  \item Compute the numerical value of \moment{\bvec{F}_T/B_N} when $F_T = 10$ N,
    $r = 1$ m, $R = 2$ m, and $\theta = \degrees{60}$.
    \\[0.45pc]
    \Solution{}{0.53\linewidth}{
      \begin{minipage}[t]{0.99\linewidth}
        \begin{tabular}{lcl}
          \moment{\bvec{F}_T/B_N}&=&$F_T(r\minus R \cos\theta)~\uvecz{n}$
          \\[1.0pc]
          &=&$10N \left(\SI{1}{m}\minus\SI{2}{m}\left(\fracDisplaystyle{1}{2}\right)\right)~\uvecz{n}$
          \\[1.0pc]
          &=&$0~\uvecz{n}$
          \\[1.0pc]
        \end{tabular}
      \end{minipage}
    }
\end{enumerate}
